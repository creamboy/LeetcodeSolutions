80.	Remove Duplicates from Sorted Array II


public class Solution {
    public int removeDuplicates(int[] nums) {
        int res=0;
        for(int i=0; i<nums.length; i++){
            if(i==nums.length-1 || i==0 || nums[i-1]!=nums[i] ||(nums[i-1]==nums[i] && nums[i+1]!=nums[i])){
                nums[res++]=nums[i];
            }
        }
        return res;
    }
}
